## Install

```bash
npm install --save masspa-ecart
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'masspa-ecart'
import 'masspa-ecart/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
